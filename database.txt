Account {
  id: INT
  username: VARCHAR(16)
  password: VARCHAR(16)
  slots: INT
  enabled: BOOL
  banned: BOOL
}

Character {
  id: INT
  account_id: INT FOREIGN KEY Account(id)
  name: VARCHAR(16)
  title: VARCHAR(12)
  birthday: DATE
  blood: INT
}

Attributes {
  character_id: INT FOREIGN KEY Character(id)
  gender: INT
  height: INT
  hair_style: INT
  hair_color: INT
  skin_color: INT
}
